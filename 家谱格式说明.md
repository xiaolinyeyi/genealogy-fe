使用json格式，支持的字段如下：
```
{
    "meta": {
        "title": "xx家谱", // 展示在页面上的标题
        "defaultCalendar": "农历" // 没有特殊说明时，生日、忌日的历法
    },
    "baseInfo": {
        "zuxun": "", // 祖训，在首页展示
        "zibei": [ // 字辈表，在首页展示，从第一代开始
            "承", "永", "祥", "熙", "铭", "泽", "恒", "昌", "嘉", "宁"
        ]
    },
    "allPeople": [ // 具体每个人的信息，以及人之间的关系
        {
            "id": 55, // 必须要有，每个人的唯一标识
            "name": "赵云",
            "children": [ // 孩子们的信息，挂在赵氏男下面，方便整理家族树
              64, // 孩子的id
              65
            ],
            "sex": true, // 性别，男为true，女为false
            "fatherID": 44, // 父亲的id，家族生人必须要有（根节点除外）。将会用于判断是否为本家
            "birthday": { // 生日信息
              "calendar": "农历", // 农历可以省略，也可以是公历
              "date": "1966-09-02T16:00:00.000Z"
            },
            "deathday": { // 忌日信息
              "calendar": "农历", // 农历可以省略，也可以是公历
              "date": "1990-12-25T16:00:00.000Z"
            },
            "nativePlace": "常山", // 户籍所在地
            "spouses": [ // 配偶信息。多次婚姻填多个。仅本家需要填
              {
                "id": 63, // 配偶id
                "time": "2014.12.22" // 结婚时间
              }
            ],
            "educational": "初中", // 文化水平
            "genID": 8, // 代数
            "habitation": [ // 居住地，可以填多个
              "常山", "成都"
            ],
            "politicalStatus": "中共党员", // 政治面貌
            "jobs": [ // 职业
              "山大王"
            ],
            "posts": [ // 职务
                "将军"
            ],
            "note": "曾七进七出", // 备注信息
            ext: [
                {
                    "birthday": "三国时期" // 不知道具体生日时间，可以在这里补充，会展示在生日栏，优先级大于外面的birthday
                }
            ]
          }
    ]
}
```
后续需要优化的：
* 日期使用更灵活的字符串
* 加入母亲信息，以适应多婚姻，同父异母
